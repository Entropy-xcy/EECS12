  1. Class Definition example
class Student:
	# __init__ is the constructor
	def __init__(self, name. hours, qpoints):
		self.name = name
		self.hours = float(hours)
		self.qpoints = float(qpoints)
	
	# The self parameter is required for a method calling self.<var>
	def getName(self):
		return self.name
	
	def getHours(self):
		return self.hours
	
	def getQPoints(self):
		return self.qpoints
		
	def gpa(self):
		return self.qpoints/self.hours
	
	# a static function which not calling self.<var> does not require self as a parameter of the method
	def makeStudent(infoStr):
		name, hours, qpoints = infoStr.split()
		return Student(name, hours, qpoints)
  1. Examples using class
>>> stu = Student("abc", 1, 1)
>>> print(stu.getName())
abc
>>> print(stu.gpa())
1
>>> stu2 = makeStudent("bcd 1 1")
>>> print(stu2.getHours())
1
